#include <iostream>
#include <algorithm>
using namespace std;

int n, c, arr[200002];
int val,p;

inline int router(int mid){
    val = arr[0]+mid;
    p = c-1;
    for(int i=1;i<n;i++){
        if (arr[i]>=val){
            p--;
            val = arr[i]+mid;
            if (p==0) return 1;
        }
    }
    return 0;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin >> n >> c;
    for(int i = 0; i<n; i++)
        cin >> arr[i];
    
    sort(arr,arr+n);
    
    int left = 1, right = (arr[n-1]-arr[0])/(c-1)+1;
    int mid,answer;
    answer = 1;
    while (left<right){
        mid = (left+right)/2;
        if (router(mid)==1){
            answer=mid;
            left = mid+1;
        }
        else right = mid;
    }
    cout<<answer;
}
